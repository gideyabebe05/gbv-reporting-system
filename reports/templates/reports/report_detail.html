from django.shortcuts import render, redirect, get_object_or_404 from django.contrib.auth import authenticate, login, logout from django.contrib.auth.decorators import login_required from django.contrib import messages from django.http import JsonResponse # ← NEW from .models import Report, Resource, Message # ← Message added from .forms import ReportForm from django.contrib.auth.models import User # 🌐 HOME / LANDING PAGE (Public - No login needed) def home(request): return render(request, 'reports/index.html') # 🔴 REPORT SUBMISSION (Public - No login needed) + Notification def report_form(request): if request.method == 'POST': form = ReportForm(request.POST) if form.is_valid(): report = form.save(commit=False) if report.is_anonymous: report.name = None report.phone = None report.save() messages.success(request, f'🔔 New report received! Tracking ID: {report.report_id}') return redirect('reports:report_success', tracking_id=report.report_id) else: form = ReportForm() return render(request, 'reports/report_form.html', {'form': form}) # 🟢 SUCCESS PAGE def report_success(request, tracking_id): report = get_object_or_404(Report, report_id=tracking_id) return render(request, 'reports/report_success.html', {'report': report, 'tracking_id': report.report_id}) # 🔍 TRACK REPORT (Public - No login needed) - NOW WITH CHAT MESSAGES def track_report(request): report = None error = None messages_list = [] if request.method == 'POST': tracking_input = request.POST.get('report_id', '').strip().upper() if tracking_input: try: report = Report.objects.get(report_id=tracking_input) messages_list = report.messages.all() except Report.DoesNotExist: try: report = Report.objects.get(tracking_id=tracking_input) messages_list = report.messages.all() except Report.DoesNotExist: error = "❌ No report found with this Tracking ID." else: error = "Please enter a Tracking ID." return render(request, 'reports/track_report.html', { 'report': report, 'error': error, 'messages': messages_list }) # 🔐 LOGIN - Supports both Email and Username def user_login(request): error = None if request.method == "POST": login_input = request.POST.get("email", "").strip() password = request.POST.get("password", "") if login_input and password: try: user_obj = User.objects.get(email=login_input) user = authenticate(request, username=user_obj.username, password=password) if user is not None: login(request, user) return redirect('reports:report_list') except User.DoesNotExist: pass user = authenticate(request, username=login_input, password=password) if user is not None: login(request, user) return redirect('reports:report_list') else: error = "Invalid email or username" else: error = "Please enter email/username and password" return render(request, 'reports/login.html', {'error': error}) # 🚪 LOGOUT @login_required(login_url='reports:login') def user_logout(request): logout(request) return redirect('reports:login') # 🏢 DASHBOARD - ONLY FOR CASE MANAGERS @login_required(login_url='reports:login') def report_list(request): reports = Report.objects.all().order_by('-created_at') role = "Admin" if request.user.is_superuser else "Case Manager" return render(request, 'reports/report_list.html', { 'reports': reports, 'role': role }) # 📄 REPORT DETAIL (Protected) - NOW WITH CHAT MESSAGES @login_required(login_url='reports:login') def report_detail(request, pk): report = get_object_or_404(Report, pk=pk) messages_list = report.messages.all() return render(request, 'reports/report_detail.html', { 'report': report, 'messages': messages_list }) # ✏️ UPDATE REPORT (Protected) @login_required(login_url='reports:login') def update_report(request, pk): report = get_object_or_404(Report, pk=pk) if request.method == 'POST': report.status = request.POST.get('status') report.notes = request.POST.get('notes') report.assigned_to = request.user report.save() messages.success(request, "Report updated successfully!") return redirect('reports:report_list') return render(request, 'reports/update_report.html', {'report': report}) # 📚 RESOURCES (Public) def resources_view(request): resources = Resource.objects.filter(is_active=True) return render(request, 'reports/resources.html', {'resources': resources}) # 📝 SIGNUP FOR NEW USERS def signup(request): error = None if request.method == "POST": username = request.POST.get("username") email = request.POST.get("email") password1 = request.POST.get("password1") password2 = request.POST.get("password2") if password1 != password2: error = "Passwords do not match" elif User.objects.filter(email=email).exists(): error = "Email already exists" elif User.objects.filter(username=username).exists(): error = "Username already exists" else: user = User.objects.create_user(username=username, email=email, password=password1) user.save() login(request, user) messages.success(request, "Account created successfully!") return redirect('reports:home') return render(request, 'reports/signup.html', {'error': error}) # =================================================================== # 🔥 TWO-WAY CHAT FUNCTIONS # =================================================================== # Case Manager sends message to Reporter @login_required(login_url='reports:login') def send_message_manager(request, pk): report = get_object_or_404(Report, pk=pk) if request.method == 'POST': text = request.POST.get('message') if text: Message.objects.create( report=report, sender_type='manager', message_text=text ) messages.success(request, 'Message sent to reporter!') return redirect('reports:report_detail', pk=pk) # Reporter sends message to Case Manager def send_message_reporter(request, report_id): report = Report.objects.filter(report_id=report_id).first() if report and request.method == 'POST': text = request.POST.get('message') if text: Message.objects.create( report=report, sender_type='reporter', message_text=text ) return redirect('reports:track_report') # 🔥 Interactive "Message this Supporter" from Resources page def request_help(request): if request.method == 'POST': import json try: data = json.loads(request.body) report_id = data.get('report_id') message_text = data.get('message_text', 'I would like to speak with a supporter. Please help me.') report = Report.objects.filter(report_id=report_id).first() if not report: return JsonResponse({'success': False, 'error': 'Report ID not found'}, status=404) Message.objects.create( report=report, sender_type='reporter', message_text=message_text ) return JsonResponse({ 'success': True, 'supporter_name': 'Supporter', 'report_id': report.report_id }) except Exception as e: return JsonResponse({'success': False, 'error': str(e)}, status=400) return JsonResponse({'success': False, 'error': 'Invalid request'}, status=400)
